Example #1

Scrape image sources from a URL.

$ phantomjs phantom-img-srcs.js <url>

http://somesite.com/image1.jpeg
http://somesite.com/image2.jpeg
http://somesite.com/image3.jpeg
http://somesite.com/image4.jpeg
http://somesite.com/image5.jpeg

Design for phantom-img-srcs.js

  1. Parse console arguments
  2. Request HTML
  3. Scrape image sources
  4. Output to console

Step 1: Parse Console Arguments

// PhantomJS provides you with the system module
var system = require('system');

// The first argument is the name of the
// script that is currently being invoked. phantom-img-srcs.js
scriptName = system.args[0];

// The second argument is the url that we will
// use to scrape image sources
url = system.args[1];

Step 2: Request The URL

// PhantomJS provides you with the webpage module
var webPage = require('webpage').create();

// Pass a url and a callback that will be invoked
// when the a response is received
webPage.open( url, function(status){

    // Status will be either 'success' or 'fail'
    // Now you can start scripting on the DOM

});

Step 3: Scrape Image Sources

// We can script on the DOM using webPage.evaluate
var imgSrcs = webPage.evaluate(function () {

    var imgs = document.querySelectorAll('img');

    // Convert nodeList to array
    imgs = Array.prototype.slice.call(imgs, 0);

    // Return array of the page's image sources out to imgSrcs
    return imgs.map(function(img){
        return img.src;
    });
});

Scope of webPage.evaluate

( 1 of 3 )

// Creates a sandboxed scripting environment

webPage.evaluate(function(){
   // Code here runs in the context of the file that was opened
   // No Access to variables in your PhantomJS script
});

Scope of webPage.evaluate

( 2 of 3 )

var outside = 2;

function doSomething(){
    return "something";
}

webPage.evaluate(function(){
    //Throws an error
    doSomething();

    // 'outside' is also not available
    var inner = outside;
});

Scope of webPage.evaluate

( 3 of 3 )

// To have access to data inside evaluate, pass it as a
// second argument

var outer = { stuff: 'stuff'};

var fromInsidePage = webPage.evaluate(function(data){

    data.stuff === 'stuff'

    return 4;
}, outer);

fromInsidePage === 4;

// Anything that can be JSONified can be passed in or returned
// No functions, DOM nodes

Step 4: Output To Console

// imgSrcs is a list of image sources from the requested page
imgSrcs.forEach(function(src){

    // This is how we output to the console
    console.log(src);
});

// phantom.exit ends our script and the PhantomJS process
phantom.exit(0);

// A successful program returns 0 and an unsuccessful one
// returns a non-zero value

Design for phantom-scrape-server.js

  1. Start a web server
  2. Parse requests
  3. Scrape image data
  4. Return JSON response

Step 1: Start A Web Server

//PhantomJS provides you with a webserver module
var server = require('webserver').create();

// We will be requiring the webpage module but will not make
// an instance until a request comes in
var WebPage = require('webpage');

// Start listening on port 8080 (localhost:8080)
service = server.listen(8080, function (request, response) {

   //here is where we will write our code to handle requests

});

Step 2: Parse Requests

// Extract the url that we will use to scrape data from
var queryString = request.url.split("?")[1];
var url = unescape(queryString.split("=")[1]);

// Create new instance of webpage to make a request
var webPage = WebPage.create();
webPage.open(url, function(status){

   // Write code to scrape the page and return a reponse

});

Step 3: Scrape Image Data

var imgData = webPage.evaluate(function(){
    var imgs = document.querySelectorAll("img");

    imgs = Array.prototype.slice.call(imgs,0);

    return imgs.map(function(img){
        return{
            src: img.src,
            alt: img.alt,
            title: img.title
        };
    });
});

Step 4: Return JSON Response

// Need to release the memory heap of the webpage
// New as of 1.7. Used to be webPage.release();
webPage.close();

// Form the response
response.statusCode = 200;
response.headers = {
    'Content-Type': "text/javascript"
};
response.write(JSON.stringify(imgData, null, 4));
response.close();

Example #3

Run javascript tests

$ phantomjs phantom-qunit.js <test-file>

Pass
Passing
Passed. You should probably drink a beer

Design for phantom-qunit.js

  1. Validate console arguments
  2. Configure webpage Instance
  3. Run QUnit test
  4. Determine test status
  5. Output results

Why QUnit?

It was my first testing framework

Step 1: Validate Console Arguments

var system = require('system');
var script = system.args[0];

if( system.args.length !== 2 ){
    console.log("Usage: phantomjs " + script + " <test-file>");
    phantom.exit(1);
}

var testFile = system.args[1];

Step 2: Configure Webpage Instance

// If there is a javascript error in the test page
// onError function will be invoked and log erros to the console
webPage.onError = function(msg, trace){
    console.log("Error on the page: " + msg);
    trace.forEach(function(item) {
        console.log('  ', item.file, ':', item.line);
    });
};

// Relay console.log calls in the test page to the console
webPage.onConsoleMessage = function(msg){
    console.log(msg);
};

// Set the size of the view port that test page will be run in
webPage.viewPortSize = { width: 980, height: 1000 };

Step 3: Run QUnit Tests

webPage.open( testFile, function(status){

    // This time let's validate the page loaded properly
    if ( status !== 'success') {
        console.log("Unable to load " + url);
        phantom.exit(2);
    }

    setInterval(function() {
        // Poll for test completion
    }, 100);

});

Step 4: Determine Test Status

//Inside setInterval so it will repeat until complete
var complete = webPage.evaluate(function () {
    var banner = document.getElementById('qunit-banner');
    var classes = banner.className;
    var failed = classes.indexOf('qunit-fail') !== -1;
    var passed = classes.indexOf('qunit-pass') !== -1;

    if( !failed && !passed ) return false;
    if(failed) return "failed";
    return "passed";
});

Example #5

Take a screen shot of every matching CSS selector.

$ phantomjs phantom-selector.js 'http://pinterest.com' '.pin'

Design for phantom-selector.js

  1. Parse console arguments
  2. Request HTML
  3. Gather element coordinates
  4. Render screen shots

Step 1: Parse Console Arguments

var system = require('system');
var webPage = require('webpage').create();
var scriptName = system.args[0];
var url = system.args[1];
var selector = system.args[2];

Step 2: Request HTML

webPage.open( url, function(status){

    // After the url has been successfully loaded
    // we want to add jQuery to it so we can use it in
    // the webPage.evaluate function
    var jQueryUrl = "http://ajax.googleapis.com/.../jquery.min.js";
    webPage.includeJs( jQueryUrl, function(){

        //jQuery has now loaded and available to use

    });
});

Step 3: Gather Element Coordinates

// Pass the css selector as the second argument
// to the webPage.evaluate function
elementData = webPage.evaluate(function (selector) {
    var $nodes = $(selector);

    // return the coordinates of the first 10 matching elements
    return $nodes.map(function(){
        var $node = $(this);
        var coords = $node.offset();
        coords.width = $node.outerWidth();
        coords.height = $node.outerHeight();
        return coords;
    }).get().slice(0,10);

}, selector);

Step 4: Render Screen Shots

elementData.forEach(function(data, i){

    // Create a unique filename for each screen shot
    var file = "selection_" + i + ".png";

    // Set the clipRect property of webPage
    // This limits the area that is rendered
    webPage.clipRect = {
        top: data.top,
        left: data.left,
        width: data.width,
        height: data.height
    };

    // Write the file to disk
    webPage.render(file);
});

phantom.exit(0);
